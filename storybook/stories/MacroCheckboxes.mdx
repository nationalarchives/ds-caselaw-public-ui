# Checkboxes Jinja Macros

Let users select one or more options by using the checkboxes component.
This page shows the Jinja macros for the Checkboxes component alongside rendered examples.

---

## Wrapper Macro

<Source
  language="jinja"
  code={`{% macro default_checkboxes() %}
<h2>Checkboxes</h2>
      {{ checkboxes_examples() }}
{% endmacro %}`}
  dark
/>

## Default Checkbox Macro

<Source
  language="jinja"
  code={`{% macro default() %}
  <h3>Default</h3>
  {{ checkboxes({
    "idPrefix": "checkbox",
    "name": "checkbox",
    "hint": {
      "text": "This is the hint text"
    },
    "items": [
      {
        "value": "option1",
        "text": "Option 1"
      },
      {
        "value": "option2",
        "text": "Option 2"
      },
      {
        "value": "option3",
        "text": "Option 3"
      }
    ]
  }) }}
{% endmacro %}`}
  dark
/>

## Error Checkbox Macro

<Source
  language="jinja"
  code={`{% macro error() %}
  <h3>With error</h3>
  {{ checkboxes({
    "idPrefix": "checkbox-with-error",
    "name": "checkbox-with-error",
    "errorMessage": {
      "text": "This is error text"
    },
    "items": [
      {

        "text": "Option 1"

## Conditional Checkbox Macro

<Source
  language="jinja"
  code={`{% macro conditional() %}
  {{ checkboxes({
    "idPrefix": "checkbox-with-conditional",
    "name": "checkbox-with-conditional",
    "items": [
      {
        "value": "option1",
        "text": "Option 1",
        "checked": true,
        "conditional": {
          "html": "This is something conditional"
        }
      },
      {
        "value": "option2",
        "text": "Option 2"
      },
      {
        "value": "option3",
        "text": "Option 3"
      }
    ]
  }) }}
{% endmacro %}`}
  dark
/>
